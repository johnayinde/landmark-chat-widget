This page demonstrates the chat widget!

Look for the floating chat button in the bottom-right corner.

The widget is embedded using just one script tag!

üéØ Method 1: Simple Embed (Easiest)

Add this single line to any website:

<script src="https://your-cdn.com/landmark-chat-widget.js"></script>

‚öôÔ∏è Method 2: With Configuration

Configure the widget using data attributes:

<script 
  src="https://your-cdn.com/landmark-chat-widget.js"
  data-api-url="https://your-api.com/api"
  data-api-key="your-api-key"
  data-title="Your Company"
  data-subtitle="Customer Support"
  data-primary-color="#ff6b6b"
  data-position="bottom-left"
  data-demo="false"
  data-auto-open="true">
</script>

üéõÔ∏è Method 3: JavaScript API

Full programmatic control:

<script src="https://your-cdn.com/landmark-chat-widget.js"></script>
<script>
  // Initialize manually
  const chatWidget = new LandmarkChatWidget({
    apiUrl: 'https://your-api.com/api',
    apiKey: 'your-api-key',
    demo: false,
    theme: {
      primaryColor: '#6366f1',
      secondaryColor: '#8b5cf6'
    },
    branding: {
      title: 'Your Company',
      subtitle: 'AI Assistant'
    },
    position: 'bottom-right',
    autoOpen: false,
    onMessage: (userMsg, botMsg) => {
      console.log('Message exchanged:', userMsg, botMsg);
      // Send to analytics
      gtag('event', 'chat_message');
    },
    onOpen: () => {
      console.log('Chat opened');
    },
    onClose: () => {
      console.log('Chat closed');
    }
  });

  chatWidget.init();

  // Control programmatically
  // chatWidget.open();
  // chatWidget.sendMessage('Hello!');
  // chatWidget.close();
</script>

üåê Platform-Specific Examples

WordPress (functions.php)

function add_landmark_chat() {
    ?>
    <script 
      src="https://your-cdn.com/landmark-chat-widget.js"
      data-title="<?php bloginfo('name'); ?>"
      data-api-url="<?php echo get_option('chat_api_url'); ?>"
      data-demo="false">
    </script>
    <?php
}
add_action('wp_footer', 'add_landmark_chat');

<script 
  src="https://your-cdn.com/landmark-chat-widget.js"
  data-title="{{ shop.name }}"
  data-primary-color="{{ settings.accent_color }}"
  data-demo="false">
</script>

HTML/Static Site

  src="https://your-cdn.com/landmark-chat-widget.js"
  data-title="My Website"
  data-demo="true">
</script>

üîß Advanced Integration

Custom triggers and interactions:

<!-- Help button opens chat -->
<button onclick="landmarkChatWidgetInstance.open()">
  Need Help?
</button>

<!-- Product inquiry -->
<button onclick="askAboutProduct('Premium Package')">
  Ask about this product
</button>

<script>
function askAboutProduct(productName) {
  landmarkChatWidgetInstance.sendMessage(
    `I'm interested in ${productName}. Can you help?`
  );
  landmarkChatWidgetInstance.open();
}

// Listen to chat events
window.addEventListener('landmark-chat-open', () => {
  gtag('event', 'chat_opened');
});
</script>

üìä Analytics Integration

<script>
const widget = new LandmarkChatWidget({
  // ... config
  onMessage: (userMsg, botMsg) => {
    // Google Analytics
    gtag('event', 'chat_message', {
      'event_category': 'Chat',
      'event_label': userMsg.content.substring(0, 50)
    });
    
    // Facebook Pixel
    fbq('track', 'Contact');
    
    // Custom analytics
    analytics.track('Chat Message', {
      message_length: userMsg.content.length,
      session_id: widget.getSessionId()
    });
  }
});
</script>
